```ts
export const getResources = async () => {
  const [provider] = await getProviders();
  if (!provider) UnexpectedCodePathError.throw('credentials required for acceptance test');

  const domains = await provider.daos.DeclaredSquarespaceDomainRegistration.get.all({}, provider.context);
  if (domains.length === 0) UnexpectedCodePathError.throw('test account must have at least one domain');

  return domains.flatMap((domain) => [
    new DeclaredSquarespaceDomainRegistration({ ...domain, isLocked: false }),
    new DeclaredSquarespaceDomainTransferRequest({ domain: refByUnique(domain) }),
  ]);
};
```
