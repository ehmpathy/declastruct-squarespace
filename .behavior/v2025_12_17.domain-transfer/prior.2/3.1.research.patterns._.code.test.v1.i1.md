## pattern: useBeforeAll for scene setup

### status: [REUSE]

## pattern: getSampleSquarespaceContext factory

### description
factory creates real Squarespace context from environment variables for integration tests.

### status: [REUSE]

### code
```ts
export const getSampleSquarespaceContext = () => {
  const email = process.env.SQUARESPACE_EMAIL;
  const password = process.env.SQUARESPACE_PASSWORD;

  if (!email || !password) {
    throw new Error('getSampleSquarespaceContext requires environment variables');
  }

  const provider = getDeclastructSquarespaceProvider({
    account: { id: process.env.SQUARESPACE_ACCOUNT_ID ?? 'integration-test', email },
    credentials: { email, password, totpSecret: process.env.SQUARESPACE_TOTP_SECRET },
  });

  return provider.context;
};
```

when('re-plan after apply to verify idempotency', () => {

    execSync(`npx declastruct plan --wish ${resourcesFile} --into ${verifyPlanFile}`, { ... });
    return { plan: JSON.parse(readFileSync(verifyPlanFile, 'utf-8')) };

  then('all resources show KEEP after apply', () => {
