```ts
export const getSquarespaceAgentOptions = (input: { ... }): ContextSquarespaceAgentOptions => {
  const browserCache = createInMemoryCache<LoggedInBrowser>({ expiration: { hours: 1 } });
  const pageCache = createInMemoryCache<Page>({ expiration: { minutes: 30 } });
  const bottleneck = new Bottleneck({ maxConcurrent: 1, minTime: 500 });
  const cacheDirectory = input.cache?.directory ?? '.cache/squarespace';

    directory: { mounted: { path: cacheDirectory } },
```

## pattern: withNewLoggedInBrowserPage wrapper

### description
higher-order function that provides a logged-in browser page to operations, handles concurrency via bottleneck.

### status: [REUSE]

### code
```ts
export const withNewLoggedInBrowserPage = <TInput, TOutput>(
  logic: (input: TInput, context: ContextSquarespaceAgentPage) => Promise<TOutput>,
  options?: { reusePageKey?: string },
) => {
  return async (input: TInput, context: ContextSquarespaceAgent): Promise<TOutput> => {
    return agentOptions.browser.bottleneck.schedule(async () => {
      const page = await getPage.execute();
      try {
        const result = await logic(input, { ...context, page });
        return result;
      } catch (error) {
        if (getPage.invalidate) getPage.invalidate({ forInput: [] });
        await page.close().catch(() => {});
        throw error;
      }
    });
  };
};
```

## pattern: getOne via getAll filter

### description
`getOneDomain` delegates to `getAllDomains` and filters in-memory rather than separate cache.

### status: [REUSE]

### code
```ts
export const getOneDomain = async (input: { by: PickOne<{ unique: ...; ref: ... }> }, context): Promise<...> => {
  const domainName = input.by.unique?.name ?? input.by.ref?.name ?? UnexpectedCodePathError.throw(...);
  const domains = await getAllDomains({}, context);
  const domainFound = domains.find((domain) => domain.name === domainName);
  return domainFound ?? null;
};
```
