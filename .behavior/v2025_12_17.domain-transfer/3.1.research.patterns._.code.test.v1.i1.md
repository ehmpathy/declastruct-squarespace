## pattern 2: useBeforeAll fixture setup

**citation [2]**: `src/access/sdks/squarespace.via.playwright/domainsList/scrapeDomainsList.integration.test.ts`

```ts
import { given, then, useBeforeAll, when } from 'test-fns';

describe('scrapeDomainsList', () => {
  given('a logged-in browser session', () => {
    const scene = useBeforeAll(async () => {
      const context = getSampleSquarespaceContext();
      const page = await getNewLoggedInBrowserPage(context.agentOptions);
      return { context, page };
    });

    afterAll(async () => {
      await scene.page.close().catch(() => {});
    });

    when('scrape of domains list page is executed', () => {
      const result = useBeforeAll(async () => {
        const domains = await scrapeDomainsList({ page: scene.page });
        return { domains };
      });

      then('returns an array of domains', async () => {
        expect(Array.isArray(result.domains)).toBe(true);
      });
    });
  });
});
```

**pattern elements:**
- `useBeforeAll` for async fixture setup
- scene object holds test context
- afterAll for cleanup
- result object holds operation output

**relation to wish:** enables shared browser session across scraper tests.

**status:** [REUSE] — pattern complete and stable

## pattern 4: sample context factory

**citation [4]**: `src/.test/getSampleSquarespaceContext.ts`

```ts
export const getSampleSquarespaceContext = () => {
  const email = process.env.SQUARESPACE_EMAIL;
  const password = process.env.SQUARESPACE_PASSWORD;

  if (!email || !password)
    UnexpectedCodePathError.throw('squarespace credentials required', {
      hint: 'set SQUARESPACE_EMAIL and SQUARESPACE_PASSWORD environment variables',
    });

  const provider = getDeclastructSquarespaceProvider({
    account: { id: process.env.SQUARESPACE_ACCOUNT_ID ?? 'integration-test', email },
    credentials: {
      email,
      password,
      totpSecret: process.env.SQUARESPACE_TOTP_SECRET,
    },
    browser: { extantBrowserWSEndpoint: process.env.BROWSER_WS_ENDPOINT },
    session: { storageStatePath: process.env.SESSION_STORAGE_PATH },
  });

  return provider.context;
};
```

**pattern elements:**
- factory returns context for test use
- credentials from environment variables
- optional browser reuse via wsEndpoint
- optional session persistence via storageStatePath

**relation to wish:** enables real browser session for integration tests.

**status:** [REUSE] — pattern complete and stable

## pattern 8: CLI workflow acceptance tests

**citation [8]**: `src/contract/sdks/declastruct.acceptance.test.ts`

```ts
describe('declastruct CLI workflow', () => {
  given('a declastruct resources file', () => {
    const testDir = join(__dirname, '.test', '.temp', 'acceptance', `run.${new Date().toISOString()}`);
    const resourcesFile = join(__dirname, '.test', 'assets', 'resources.acceptance.ts');
    const planFile = join(testDir, 'plan.json');

    when('plan is generated via declastruct CLI', () => {
      const prep = useBeforeAll(async () => {
        execSync(`npx declastruct plan --wish ${resourcesFile} --into ${planFile}`,
          { stdio: 'inherit', env: process.env });
        return {
          plan: JSON.parse(readFileSync(planFile, 'utf-8')) as { changes: DeclastructChange[] },
        };
      });

      then('creates a valid plan file', () => {
        expect(existsSync(planFile)).toBe(true);
        expect(prep.plan).toHaveProperty('changes');
      });

      then('fetches real domains from live account', () => {
        expect(prep.plan.changes.length).toBeGreaterThan(0);
      });
    });
  });
});
```

**pattern elements:**
- execSync for CLI execution
- temp directory per run
- parse plan JSON for assertions
- verify real data fetched

**relation to wish:** validates full declastruct workflow with real Squarespace account.

**status:** [REUSE] — pattern complete and stable

when('plan is regenerated after apply to verify idempotency', () => {
  const verifyPlanFile = join(testDir, 'plan-verify.json');

  const prep = useBeforeAll(async () => {

## pattern 10: resources acceptance file

**citation [10]**: `src/contract/sdks/.test/assets/resources.acceptance.ts`

```ts
export const getResources = async () => {
  const [provider] = await getProviders();
  if (!provider)
    UnexpectedCodePathError.throw('provider required for acceptance test');

  const domains = await provider.daos.DeclaredSquarespaceDomainRegistration.get.all({}, provider.context);

  if (domains.length === 0)
    UnexpectedCodePathError.throw('test account must have at least one domain', {
      hint: 'add a domain to the test Squarespace account',
    });

  return domains.flatMap((domain) => [
    new DeclaredSquarespaceDomainRegistration({
      ...domain,
      isLocked: false,
      dnssecEnabled: false,
    }),
    new DeclaredSquarespaceDomainTransferRequest({
      domain: refByUnique<typeof DeclaredSquarespaceDomainRegistration>(domain),
      requestedAt: null,
      status: 'REQUESTED',
    }),
  ]);
};
```

**pattern elements:**
- getProviders() for provider config
- getResources() for resource declarations
- fail-fast if no domains
- uses refByUnique for typed refs

**relation to wish:** enables declastruct CLI to test real workflow.

**status:** [REUSE] — pattern complete and stable
