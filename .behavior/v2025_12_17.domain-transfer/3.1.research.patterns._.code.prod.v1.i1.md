const getAllDomainsFromPage = async (_input, context: ContextSquarespaceAgentPage) => {

## pattern 4: remote state cache

export const { withRemoteStateQueryCache, withRemoteStateMutationRegistration } =
  createRemoteStateCacheContext<[any, ContextSquarespaceAgent], SimpleOnDiskCache>({

## pattern 5: browser page wrapper

**citation [5]**: `src/access/sdks/squarespace.via.playwright/wrappers/withNewLoggedInBrowserPage.ts`

```ts
export const withNewLoggedInBrowserPage = <TInput, TOutput>(
  logic: (input: TInput, context: ContextSquarespaceAgentPage) => Promise<TOutput>,
  options?: { reusePageKey?: string },
) => {
  return async (input: TInput, context: ContextSquarespaceAgent): Promise<TOutput> => {
    return agentOptions.browser.bottleneck.schedule(async () => {
      const page = await getPage.execute();
      const sessionStatus = await checkSessionHealth(page);
      if (!sessionStatus.valid) {
        await performSquarespaceLogin(page, agentOptions.credentials);
      }
      try {
        const result = await logic(input, { ...context, page });
        const randomDelay = Math.floor(Math.random() * 300) + 200;
        await new Promise((fulfill) => setTimeout(fulfill, randomDelay));
        return result;
      } catch (error) {
        if (getPage.invalidate) getPage.invalidate({ forInput: [] });
        throw error;
      }
    });
  };
};
```

**pattern elements:**
- wraps logic with browser page lifecycle
- bottleneck for sequential execution (rate limit)
- session health check before each operation
- auto re-login on session expiry
- random delay for bot detection mitigation
- page cache invalidation on error

**relation to wish:** provides resilient page management for batch operations.

**status:** [REUSE] — pattern complete and stable

## pattern 6: scraper functions

**citation [6]**: `src/access/sdks/squarespace.via.playwright/domainsList/scrapeDomainsList.ts`

```ts
export const scrapeDomainsList = async (input: { page: Page }): Promise<RawDomainListItem[]> => {
  const { page } = input;

  // navigate if needed
  if (!currentUrl.includes('/config/domains')) {
    await page.goto('https://account.squarespace.com/domains');
    await page.waitForLoadState('networkidle');
  }

  // check for empty state
  const emptyState = await page.$(domainsListSelectors.emptyState);
  if (emptyState) return [];

  // scroll to load all
  await scrollToLoadAllDomains({ page });

  // extract data from rows
  const domainRows = await page.$$(domainsListSelectors.domainRow);
  const domains = [];
  for (const row of domainRows) {
    const name = await row.$(domainsListSelectors.domainName).textContent();
    domains.push({ name, status, expiryText });
  }
  return domains;
};
```

**pattern elements:**
- input is page only (no context)
- navigate if not on target page
- check for empty state
- scroll for lazy-loaded content
- extract raw data via selectors
- return raw shape (not domain object)

**relation to wish:** extracts raw data from squarespace UI.

**status:** [REUSE] — pattern complete and stable

## pattern 9: captcha answer

**citation [9]**: `src/access/sdks/squarespace.via.playwright/captcha/answerCaptchaChallenge.ts`

```ts
export const answerCaptchaChallenge = async (input: {
  page: Page;
  handle: BrowserAuthSession;
  captchaConfig: CaptchaConfig;
}, context) => {
  const detection = await detectCaptchaChallenge({ page });
  if (!detection.detected) return { solved: true, method: 'none' };

  // tier 2: headful checkbox click
  if (captchaConfig.checkboxClickEnabled !== false) {
    await handle.respawn({ mode: 'HEADFUL' });
    const headfulPage = await handle.context.newPage();
    const clickResult = await clickTurnstileCheckbox({ page: headfulPage });
    if (clickResult.solved) {
      await handle.storage.set();
      await handle.respawn({ mode: 'HEADLESS' });
      return { solved: true, method: 'checkbox' };
    }
  }

  // tier 3: human fallback
  if (captchaConfig.humanFallbackEnabled !== false) {
    await waitForHumanSignal({ signalMode: captchaConfig.humanSignalMode ?? 'stdin' });
    await handle.storage.set();
    await handle.respawn({ mode: 'HEADLESS' });
    return { solved: true, method: 'human' };
  }

  return { solved: false, method: 'none' };
};
```

**pattern elements:**
- tiered approach: detect → checkbox → human
- browser respawn for headful mode
- session preserve across respawn
- configurable via captchaConfig

**relation to wish:** enables batch operations despite bot detection.

**status:** [REUSE] — pattern complete and stable
