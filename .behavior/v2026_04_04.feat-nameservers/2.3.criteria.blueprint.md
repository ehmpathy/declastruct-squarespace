- usecase.1 = get nameservers ✓
- usecase.2 = set custom nameservers (swap to cloudflare) ✓
- usecase.3 = reset to squarespace default (swap back) ✓
- usecase.4 = validation edgecases ✓
- usecase.5 = findsert semantics ✓

---

## subcomponent contracts

```
given('DeclaredSquarespaceDomainNameservers domain object')
  then('extends DomainEntity')
  then('has domain: RefByUnique<typeof DeclaredSquarespaceDomainRegistration>')
  then('has nameservers: string[] | null')
  then('unique key is [domain.name]')

given('getNameservers operation')
  then('exposes: getNameservers({ by: { unique: { domain: { name } } } }, context) => DeclaredSquarespaceDomainNameservers | null')
  then('returns null when domain has squarespace default nameservers')
  then('returns entity with nameservers array when custom')

given('setNameservers operation')
  then('exposes: setNameservers({ upsert: DeclaredSquarespaceDomainNameservers }, context) => DeclaredSquarespaceDomainNameservers')
  then('exposes: setNameservers({ findsert: DeclaredSquarespaceDomainNameservers }, context) => DeclaredSquarespaceDomainNameservers')
  then('validates nameservers has at least 2 entries when non-null')
  then('validates nameserver FQDN format')
  then('treats empty array as null')

given('DeclaredSquarespaceDomainNameserversDao')
  then('created via genDeclastructDao')
  then('exposes: get.one.byUnique')
  then('exposes: set.findsert')
  then('exposes: set.upsert')
  then('set.delete is null (not supported)')
```
